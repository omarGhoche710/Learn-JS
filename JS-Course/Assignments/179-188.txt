
function getArticles(apiLink) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.open("GET", apiLink);
    xhr.send();

    xhr.onload = function () {
      if (xhr.status === 200) {
        resolve(JSON.parse(xhr.responseText));
      } else {
        reject(new Error("Server Failed"));
      }
    };

    xhr.onerror = function () {
      reject(new Error("Network Failed"));
    };
  });
}

async function run() {
  try {
    const articles = await getArticles('articles.json');
    articles.length = 5;

    articles.forEach(article => {
      document.body.appendChild(createArticle(article));
    });

  } catch (err) {
    console.log(err);
  }
}

run(getArticles);
