function addClass(array) {
  let spans = document.querySelectorAll("#result span");
  let result = document.getElementById("result");
  if (spans.length === 0) {
    result.innerHTML = "";
  }

  array.forEach((element) => {
    let span = document.createElement("span");

    span.textContent = element;
    span.className = element;
    span.style.cssText =
      "padding: 10px; background-color: #ff9800; color: white; border-radius: 3px;";

    result.appendChild(span);
  });
}

function removeClass(array) {
  array.forEach((element) => {
    let spans = document.getElementsByClassName(element);
    spans = Array.from(spans);
    if (spans.length !== 0) {
      spans.forEach((ele) => ele.remove());
    } else {
      // Swal.fire({
      //   title: "Error!",
      //   text: `The Class ${element} Is Not Found`,
      //   icon: "error",
      //   confirmButtonText: "Ok",
      // });
      document.querySelector(".errors").classList.remove("hidden");
      document.querySelector(".errors").innerHTML +=
        `Class ${element} is not Found<br>`;
    }
  });
  let result = document.querySelector("#result");
  if (result.innerHTML === "") {
    result.innerHTML = "No Classes To Show";
  }
}

let inputAdd = document.getElementById("add");
let inputRemove = document.getElementById("remove");

inputAdd.addEventListener("blur", function () {
  if (this.value === "") {
    Swal.fire({
      title: "Error!",
      text: "Please Enter Class",
      icon: "error",
      confirmButtonText: "Ok",
    });
  } else {
    document.querySelector(".errors").innerHTML = "";
    document.querySelector(".errors").classList.add("hidden");
    let classes = this.value.split(" ");
    addClass(classes);
    this.value = "";
  }
});

inputRemove.addEventListener("blur", function () {
