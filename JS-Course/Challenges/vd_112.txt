let mySelectColor = document.querySelector('.select-color');
let myColors = document.querySelectorAll(".select-color div");
myMain = document.querySelector('.main');


myColors.forEach(e => {
  e.addEventListener('click', function() {
    myColors.forEach(e => {
      e.classList.remove('active');
    });
    
    window.localStorage.setItem('color', this.classList.item(0));
    myMain.style.backgroundColor = window.localStorage.getItem('color');
    this.classList.add('active');
  });
});
