    - The new Object is related by the prototype, that mean if i change in

Big Example:

const cartItemProto = {
  render() {
    this.el.textContent = `${this.name} - ${this.price}$ (x${this.qty})`;
  },
  addToCart(container) {
    container.append(this.el);
  },
  increaseQty() {
    this.qty++;
    this.render();
  }
};

function createCartItem(name, price) {
  return Object.create(cartItemProto, {
    el: {
      value: document.createElement("div")
    },
    name: {
      value: name,
      writable: true
    },
    price: {
      value: price,
      writable: true
    },
    qty: {
      value: 1,
      writable: true
    }
  });
}

let main = document.getElementById("main");

let item = createCartItem("HP Laptop", 350);

item.render();
item.addToCart(main);
