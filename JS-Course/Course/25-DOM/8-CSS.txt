function highlightElement(element) {
    // حفظ الأنماط الأصلية
    const originalStyles = element.style.cssText;
    
    // تطبيق highlight
    element.style.cssText += 'background: yellow; transition: all 0.3s;';
    
    // إزالة بعد 2 ثانية
    setTimeout(() => {
        element.style.cssText = originalStyles;
    }, 5000);
}

highlightElement(document.querySelector('div'))
