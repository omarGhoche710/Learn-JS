JS:
    let btn = document.getElementById("display");
    let amount = 0;
    let nb = [0, 0, 0, 0];

    function total(val, btn, index) {
      let value = parseFloat(document.getElementById(val).textContent);
      let button = document.getElementById(btn);
      if(nb[index] % 2 == 0) {
        amount += value;
      }else {
        amount -= value;
        console.log(`index[${index}]: ${nb[index]}`);
      }
      nb[index]++;

      button.classList.toggle('click');
    }

    btn.addEventListener("click", ()=>{
      // let result = `Your amount is ${amount.toFixed(2)}`;
      let result = `Your amount is ${Math.round(amount)}`;
      document.getElementById("result").innerHTML = result;
    });
