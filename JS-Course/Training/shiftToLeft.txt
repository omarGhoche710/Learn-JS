// The left shift operation is similar to multiplication by powers of two.
// Ex:
      10 << 3 = 10 * 2^3 = 10 * 8 = 80
      -32 << 2 = -32 * 2^2 = -32 * 4 = -128
      5 << 2 = 5 * 2^2 = 5 * 4 = 20

function shiftToLeft(x, y) {
	if (y < 0) {
		return "the power number should be positive";
	}
	return x * (2 ** y);
}
console.log(shiftToLeft(10, 3)); // 80
